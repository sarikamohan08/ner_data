import anthropic

API_KEY = "your_actual_anthropic_api_key"
client = anthropic.Anthropic(api_key=API_KEY)

try:
    response = client.messages.create(
        model="claude-3.5-sonnet-20240620-v2",
        max_tokens=50,
        temperature=0,
        messages=[{"role": "user", "content": "Hello, Claude! How are you today?"}]
    )
    print(response.content[0].text)  # Print response from Claude
except Exception as e:
    print(f"Error: {e}")
