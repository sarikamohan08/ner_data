import re
import os
import pandas as pd

# Define the regular expression pattern
pattern = r'(?:effective date of|commence on)\s*(\d{1,2}[./]\d{1,2}[./]\d{2,4}|\w+\s\d{1,2},\s\d{4})'

# Define the path to the folder containing text files
folder_path = 'path_to_your_folder'  # Replace this with your folder path

# List to store data for the Excel sheet
data = []

# Iterate through all text files in the folder
for filename in os.listdir(folder_path):
    if filename.endswith('.txt'):  # Process only .txt files
        file_path = os.path.join(folder_path, filename)

        # Open and read the contents of the file
        with open(file_path, 'r', encoding='utf-8') as file:
            text = file.read()

            # Find all matches in the text
            matches = re.findall(pattern, text, re.IGNORECASE)

            # Append file name and matched dates to the data list
            if matches:
                for date in matches:
                    data.append([filename, date])

# Convert the data into a pandas DataFrame
df = pd.DataFrame(data, columns=['File Name', 'Extracted Date'])

# Write the DataFrame to an Excel file
output_file = 'extracted_dates.xlsx'  # Output Excel file name
df.to_excel(output_file, index=False, engine='openpyxl')

print(f"Extracted dates have been written to {output_file}")
