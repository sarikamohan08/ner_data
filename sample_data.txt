import fitz  # PyMuPDF
from pymupdf4llm import to_markdown

# Step 1: Load the PDF document
pdf_path = "example.pdf"  # Replace with your PDF file path
doc = fitz.open(pdf_path)

# Step 2: Extract text and convert to Markdown
markdown_text = ""
for page_num in range(len(doc)):
    page = doc.load_page(page_num)
    markdown_text += to_markdown(page.get_text("text"))

# Step 3: Write the Markdown content to a text file
output_md_path = "output.md"  # Output file path
with open(output_md_path, "w", encoding="utf-8") as md_file:
    md_file.write(markdown_text)

print(f"Markdown content has been written to {output_md_path}")
