   import PyPDF2  # or import pdfplumber

   def extract_text_from_pdf(pdf_path):
       try:
           with open(pdf_path, 'rb') as file:
               reader = PyPDF2.PdfReader(file)  # or pdfplumber.open(pdf_path)
               text = ""
               for page_num in range(len(reader.pages)):
                   page = reader.pages[page_num]  # or reader.pages[page_num]
                   text += page.extract_text()  # or page.extract_text()
               return text
       except Exception as e:
           print(f"Error: {e}")
           return None
