from openpyxl import load_workbook

def rename_excel_sheets_inplace(file_path):
    """
    Rename specific sheets in an Excel file (in-place) using a predefined case-insensitive map.

    Parameters:
    - file_path (str): Path to the Excel file to be modified.
    """
    # Predefined renaming map (case-insensitive)
    rename_map = {
        'bot output-reply': 'ocr output-reply',
        'bot output-rts': 'ocr output-rts'
    }

    # Convert keys to lowercase for comparison
    lower_rename_map = {k.lower(): v for k, v in rename_map.items()}

    # Load the workbook
    wb = load_workbook(filename=file_path)

    # Perform renaming
    for sheet_name in wb.sheetnames:
        sheet_name_lower = sheet_name.lower()
        if sheet_name_lower in lower_rename_map:
            ws = wb[sheet_name]
            ws.title = lower_rename_map[sheet_name_lower]

    # Overwrite original file
    wb.save(file_path)
    print(f"Workbook successfully updated: {file_path}")
