import os

file_path = 'file.png'

if os.path.exists(file_path):
    if os.access(file_path, os.W_OK):
        try:
            os.remove(file_path)
        except OSError as e:
            print(f"Error removing file: {e}")
    else:
        print(f"No write permissions for {file_path}")
else:
    print(f"File {file_path} does not exist")
