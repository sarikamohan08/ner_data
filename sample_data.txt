from selenium import webdriver
from selenium.webdriver.edge.service import Service
from webdriver_manager.microsoft import EdgeChromiumDriverManager
from selenium.webdriver.common.by import By
import time

# Step 1: Setup Edge WebDriver
driver = webdriver.Edge(service=Service(EdgeChromiumDriverManager().install()))

# Step 2: Open the Website
url = "https://example.com"  # Replace with the target URL
driver.get(url)

# Step 3: Locate and Click the Export Button
export_button = driver.find_element(By.CLASS_NAME, "export-button-class")  # Replace with the actual class name
export_button.click()

# Step 4: Wait for the File to Download
time.sleep(5)  # Adjust based on download time

# Step 5: Clean up
driver.quit()
