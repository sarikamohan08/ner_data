def read_specific_line(file_path, line_number=308):
    with open(file_path, 'r') as f:
        for current_line_number, line in enumerate(f, start=1):
            if current_line_number == line_number:
                return line.strip()  # Returns the text of line 308
    return None  # If the file has fewer than 308 lines

# Example usage:
line_308 = read_specific_line('your_file.txt')
print(line_308)
