def append_to_excel(df, excel_file_path):
    try:
        # Check if the file exists
        if os.path.exists(excel_file_path):
            # If the file exists, append the DataFrame without writing headers again
            with pd.ExcelWriter(excel_file_path, engine='openpyxl', mode='a') as writer:
                # Find the last row of the existing sheet
                df.to_excel(writer, sheet_name='ExtractedData', startrow=writer.sheets['ExtractedData'].max_row, header=False, index=False)
        else:
            # If the file doesn't exist, create it and add the DataFrame with headers
            with pd.ExcelWriter(excel_file_path, engine='openpyxl') as writer:
                df.to_excel(writer, sheet_name='ExtractedData', index=False)

    except Exception as e:
        print(f"Error appending to Excel: {e}")
