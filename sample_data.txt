def set_indicators(data):
    for key, value in data.items():
        # Handle case where value is a list of dicts
        if isinstance(value, list):
            for entry in value:
                # Check if extracted text exists and is non-empty
                extracted_text = entry.get('extracted text from pdf', '')
                entry[f"{key}_ind"] = 1 if extracted_text else 0
        # Handle case where value is a dict
        elif isinstance(value, dict):
            extracted_text = value.get('extracted text from pdf', '')
            value[f"{key}_ind"] = 1 if extracted_text else 0
        # For other types, optionally skip or handle as needed
    return data
