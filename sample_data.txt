def extract_text_from_doc(doc_path):
    # Load the DOC/DOCX file
    doc = Document(doc_path)

    # Extract text from each paragraph
    full_text = ""
    for para in doc.paragraphs:
        full_text += para.text + "\n"

    return full_text
