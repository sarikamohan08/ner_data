import os
import pandas as pd

def extract_file_paths_to_excel(root_folder, output_excel):
    # List to store file paths
    file_paths = []

    # Traverse the directory tree
    for dirpath, dirnames, filenames in os.walk(root_folder):
        for file in filenames:
            file_paths.append(os.path.join(dirpath, file))

    # Create a DataFrame from the file paths
    df = pd.DataFrame(file_paths, columns=["File Path"])

    # Save the DataFrame to an Excel file
    df.to_excel(output_excel, index=False)

    print(f"File paths extracted and saved to {output_excel}")

# Example usage
root_folder = "/path/to/your/root/folder"  # Replace with your root folder path
output_excel = "file_paths.xlsx"          # Output Excel file name
extract_file_paths_to_excel(root_folder, output_excel)
