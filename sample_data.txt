def create_grouper_indicator(self, description):
        """
        Checks if the word 'grouper' is present in the description.

        Args:
            description (str): The text to check for the keyword.

        Returns:
            int: 1 if 'grouper' is found, otherwise 0.
        """
        if not isinstance(description, str):
            return 0
        return 1 if 'grouper' in description.lower() else 0
