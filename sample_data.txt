def chunk_text(text, chunk_size=2000, overlap=200):
    """
    Splits text into overlapping chunks for LLM processing.
    - text: Input string.
    - chunk_size: Max token/word length per chunk (roughly, since it's word-based).
    - overlap: Number of words overlapping between chunks (captures boundary entities).
    Returns: List of text chunks.
    """
    words = text.split()
    chunks = []
    i = 0
    while i < len(words):
        # Build chunk and make sure last chunk is taken
        chunk = words[i:i+chunk_size]
        chunks.append(' '.join(chunk))
        if i + chunk_size >= len(words):
            break
        i += chunk_size - overlap
    return chunks

# Example usage:
full_contract_text = "..."  # your long contract text here
chunks = chunk_text(full_contract_text, chunk_size=2000, overlap=200)
print(f"Number of chunks: {len(chunks)}")
