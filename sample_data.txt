def rename_sheets_in_excel(file_path):
    # Load the Excel file
    xls = pd.ExcelFile(file_path)

    # Mapping with lowercase keys for case-insensitive matching
    rename_map = {
        'bot  output reply': 'ocr  output reply',
        'bot output rts': 'ocr output rts'
    }

    # Write the modified file back to the same path
    with pd.ExcelWriter(file_path, engine='openpyxl') as writer:
        for sheet_name in xls.sheet_names:
            df = xls.parse(sheet_name)
            # Normalize to lowercase and strip spaces for matching
            normalized_name = sheet_name.lower().strip()
            new_name = rename_map.get(normalized_name, sheet_name)
            df.to_excel(writer, sheet_name=new_name, index=False)

    return file_path
