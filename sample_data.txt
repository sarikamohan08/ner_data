reverse_dict = {v: k for k, v in tax_provider_dict.items()}

# Function to find tax_id based on provider_name
def find_tax_id(name):
    for full_name, tax_id in reverse_dict.items():
        if name in full_name:
            return tax_id
    return None

# Apply the function to the DataFrame
df['tax_id'] = df['provider_name'].apply(find_tax_id)
