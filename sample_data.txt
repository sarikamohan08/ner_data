# Step 6: Wait for the modal to disappear before clicking the "Export" button again
    WebDriverWait(driver, 10).until(
        EC.invisibility_of_element_located((By.CSS_SELECTOR, "div[role='dialog'][aria-modal='true']"))
    )

    # Step 7: Click the "Export" button again to initiate the download
    export_button_again = WebDriverWait(driver, 10).until(
        EC.element_to_be_clickable((By.CLASS_NAME, "export_btn"))
    )
    
    # Scroll the Export button into view again
    driver.execute_script("arguments[0].scrollIntoView();", export_button_again)
    
    # Click the "Export" button to download the zip file
    export_button_again.click()
